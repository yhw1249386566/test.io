# 选择排序

```js
function selectSort(arr) {
  let minIndex;
  let temp;

  for (let i = 0; i < arr.length; i++) {
    minIndex = i;

    // j+1: 因为 i 前面的都排序好了, 如果使用 j=0, 那么 minIndex 永远会等于最前面的索引, 即: 0
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[j] < arr[minIndex]) {
        minIndex = j;
      }
    }

    // 将最小值交换到前面
    temp = arr[i];
    arr[i] = arr[minIndex];
    arr[minIndex] = temp;
  }

  return arr;
}

const sourceArr = [32, 12, 56, 78, 76, 45, 36];
const result = selectSort(sourceArr);

console.log(result); // [12, 32, 36, 45, 56, 76, 78]
```

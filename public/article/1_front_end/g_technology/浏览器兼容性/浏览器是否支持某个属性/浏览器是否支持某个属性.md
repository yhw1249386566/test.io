# 当前浏览器是否支持某个属性

```js
/** 2020/4/1 */
<script type="text/javascript">
    function demo() {
        var div = document.createElement('div'),
            vendors = 'Ms O Moz Webkit'.split(' '),
            len = vendors.length;

        return function (prop) {
            if (prop in div.style) return true;

            prop = prop.replace(/^[a-z]/,function (val) {
                return val.toUpperCase();
            });

            while (len--) {
                if (vendors[len] + prop in div.style) {
                    return true;
                }
            }
            return false;
        };
    }

    var support_css3 = (demo)();

    // 若设备不支持xx属性
    if (!support_css3('grid')) {
        let text = '您当前的浏览器版本太低,请更换或升级您的浏览器!',
            h1 = document.createElement("h1"),
            h1Content = document.createTextNode(text),
            firstBody;
        // 添加文本节点 到这个新的div元素. W
        h1.appendChild(h1Content);
        h1.setAttribute('id', 'support display h1')
        firstBody = document.getElementsByTagName('body')[0]
        firstBody.innerHTML = '';
        firstBody.appendChild(h1).style.fontSize = '30px'
    }
</script>
```

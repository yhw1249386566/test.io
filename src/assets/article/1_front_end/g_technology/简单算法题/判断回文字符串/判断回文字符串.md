const str = "$ab  $c,,b,...a";

function isValid(char: string) {
  if ((char >= "a" && char <= "z") || (char >= "0" && char <= "9")) {
    return true;
  }

  return false;
}

function isPalindrome(str: string) {
  // 双指针，一个指向左，一个指向右
  let left = 0,
    right = str.length - 1;

  // 循环：让左右指针一个一个字符的移动，
  // 当右指针超过/等于左指针，说明它们已经遍历完整个字符串了
  while (right >= left) {
    const lv = str[left].toLowerCase();
    const rv = str[right].toLowerCase();

    if (!isValid(lv)) {
      left++;
      continue;
    }

    if (!isValid(rv)) {
      right--;
      continue;
    }

    if (lv !== rv) {
      return false;
    }

    left++, right--; // 如果此时两字符相等，就进行下一轮比较
  }

  // 都通过了，说明是回文字符串
  return true;
}

console.log(isPalindrome(str)); // true
